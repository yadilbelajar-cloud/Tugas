async function sendToAI() { const promptInput = document.getElementById('promptInput').value; const output = document.getElementById('responseOutput'); const resultSection = document.getElementById('resultSection'); if (!promptInput) return; try { const response = await fetch('/api/generate', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ prompt: promptInput }) }); // 1. Ambil data sebagai JSON const data = await response.json(); // 2. Cek di Console (Tekan F12 di browser saat tes) console.log("Data dari Server:", data); // 3. Validasi sebelum menampilkan if (data && data.text) { output.innerText = data.text; } else { output.innerText = "Error: Format data tidak sesuai (Undefined)."; } resultSection.classList.remove('hidden'); } catch (error) { output.innerText = "Koneksi gagal: " + error.message; resultSection.classList.remove('hidden'); } }
